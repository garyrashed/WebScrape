    function toggleItem() {
	
	    // Closes all accordions 
        var itemClass = this.parentNode.className;
	
	    // Counter to close every accordion

          item[i].className = 'dropdown-secondary close';

        addEvent: function (obj, ev, handler) {
            if (!obj) return;

            if (obj.addEventListener)
                obj.addEventListener(ev, handler, false);
            else if (obj.attachEvent)
                obj.attachEvent('on' + ev, handler);
        },

        SelectPropertyType: function (categoryname, categoryId, searchtype) {

            if (categoryname == '' && searchDisplayType == 'MultiLevel') {
                $(this.DropDownID).innerHTML = 'Select a Property Type';
                $(this.DropDownID).style.color = '#999';
                return;
            }

            $(this.DropDownID).style.color = '#466daa';
            if (searchDisplayType == 'MultiLevel') { $('multiLevelOptionsWrapper_ErrorAlert').style.display = 'none'; }
            var valueHasChanged = false;

            if (
                searchtype != UniversalQuickSearch.SearchType ||
                categoryId != UniversalQuickSearch.CategoryID) {
                valueHasChanged = true;
            }

            this.CategoryName = categoryname;
            var selectedItemDisplay = categoryname + '<span class="caret-down-small"></span>';

            if (searchtype == "FS" && searchDisplayType == 'MultiLevel' && categoryId != '') {
                selectedItemDisplay += ' For Sale';
            }
            if (searchtype == "FL" && searchDisplayType == 'MultiLevel' && categoryId != '') {
                selectedItemDisplay += ' For Lease';
            }            

            // hack to trim the name so it looks nice in the textbox
            if (searchtype == "FS" && categoryId == 130 && searchDisplayType == 'MultiLevel') {
                selectedItemDisplay = 'Sport & Ent. For Sale';
            }

            $(this.DropDownID).innerHTML = selectedItemDisplay;

            UniversalQuickSearch.SearchType = searchtype;
            UniversalQuickSearch.CategoryID = categoryId;

            hidePropertyTypes();

            if (valueHasChanged && this.onchange && typeof (this.onchange) == "function") {
                this.onchange();  // probably not the best way to "fire an event"
            }
        },



        DoSearch: function (event) {
            if ($(this.DropDownID).innerHTML.indexOf("Select a Property Type") > -1) {
                UniversalQuickSearch.TempDontHide = true;
                if ($(activeOptionList).style.display == 'none') {
                    UniversalQuickSearch.togglePropertyTypesFromLink();
                }
                else {
                    $(this.DropDownID).focus();
                }
                if (searchDisplayType == 'MultiLevel') { $('multiLevelOptionsWrapper_ErrorAlert').style.display = 'block'; }

                setTimeout("UniversalQuickSearch.TempDontHide=false;", 203);
                LogActionCode(1130, PageExecutionGuid);
                return false;
            }

            var QS_Form = document.createElement("FORM");
            document.body.appendChild(QS_Form);

            var FromUniversalInput = document.createElement("INPUT");
            FromUniversalInput.type = "hidden";
            FromUniversalInput.name = "FromUniversalSearch";
            FromUniversalInput.value = "true";

            var QuickSearchBlurString = document.createElement("INPUT");
            QuickSearchBlurString.type = "hidden";
            QuickSearchBlurString.name = "UniversalSearchBlurString";
            QuickSearchBlurString.value = UniversalQuickSearch.BlurString;

            var LocationInput = document.createElement("INPUT");
            LocationInput.type = "hidden";
            LocationInput.name = "UniversalSearchLocation";
            LocationInput.value = UniversalQuickSearch.LocationInputField.value;

            var PropertyTypeInput = document.createElement("INPUT");
            PropertyTypeInput.type = "hidden";
            PropertyTypeInput.name = "UniversalSearchPropertyType";
            PropertyTypeInput.value = UniversalQuickSearch.CategoryID;

            var doSearchLinkCode = "";
            var linkCodeQS = "";

            if (parseInt(doSearchLinkCode) > 0)
                linkCodeQS = "?linkcode=" + doSearchLinkCode;

            if (UniversalQuickSearch.SearchType == "FS") {
                QS_Form.action = "/forsale/" + linkCodeQS;
            }
            else if (UniversalQuickSearch.SearchType == "FL") {
                QS_Form.action = "/forlease/" + linkCodeQS;
            }

            QS_Form.appendChild(PropertyTypeInput);
            QS_Form.appendChild(LocationInput);
            QS_Form.appendChild(FromUniversalInput);
            QS_Form.appendChild(QuickSearchBlurString);
            QS_Form.method = "POST";

            QS_Form.submit();

            return false;
        },

                setTimeout("UniversalQuickSearch.TempDontHide=false;", 203);
            }
        }
    });

    }

    function toggleTabs(element, activeClass, desiredTag,searchType,optionList) {
        var elements = element.parentNode.parentNode.getElementsByTagName(desiredTag);

        for (var i = 0; i < elements.length; i++)
            elements[i].className = "";

        element.className = activeClass;
        activeTabSearchType = searchType;
        activeOptionList = optionList;

        $('TabbedFSOptions').style.display = 'none';
        $('TabbedFLOptions').style.display = 'none';

        switch (searchType) {
            case "FS":
                setCorrespondingListItem(FSCategories, searchType, 'All properties for sale');
                break;
            case "FL":
                setCorrespondingListItem(FLCategories, searchType, 'All properties for lease');
                break;
            default:
        }
    }

    function setCorrespondingListItem(categoryIDArray, searchType, defaultCategoryName)
    //select an item based on previous list for Tabbed QuickSearch
    {
        var len = categoryIDArray.length;
        var itemFound = false;
        for (var i = 0; i < len; i++) {
            if (categoryIDArray[i] == UniversalQuickSearch.CategoryID) //found
            {
                itemFound = true;
                UniversalQuickSearch.SelectPropertyType(UniversalQuickSearch.CategoryName, categoryIDArray[i], searchType);
                break;
            }
        }

        if (!itemFound) {
            UniversalQuickSearch.SelectPropertyType(defaultCategoryName, '', searchType);
        }
    }
